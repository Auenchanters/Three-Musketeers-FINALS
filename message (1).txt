E. New helper: extract observed (service, effect) candidates.
File: web/training_loop.py.
Add after _candidate_actions (~line 127):

# Canonical effect-string vocabulary — the union of every effect token used
# by the seed_generator chain templates (data/seed_generator.py:31-73). Each
# entry is a substring keyword we look for in log messages; matching is
# case-insensitive on the underscore form. We deliberately keep the mapping
# explicit (not regex-built from seed_generator) so adding a new chain
# template forces a conscious update here.
_EFFECT_KEYWORDS: dict[str, tuple[str, ...]] = {
    "connection_pool_exhaustion": ("connection pool", "pool exhausted"),
    "upstream_timeout":           ("upstream", "504", "gateway timeout"),
    "5xx_errors_to_users":        ("5xx", "503", "returned to client"),
    "oom_crash_loop":             ("oom", "out of memory", "crash loop"),
    "dependency_timeout":         ("dependency", "dependent service"),
    "service_degradation":        ("degraded", "degradation"),
    "complete_unavailability":    ("unavailable", "unreachable"),
    "resource_limit_exceeded":    ("limit exceeded", "quota"),
    "cascading_timeout":          ("cascad", "cascade"),
    "user_facing_errors":         ("user-facing", "user facing"),
    "network_degradation":        ("network", "packet loss"),
    "failover_triggered":         ("failover", "fail-over"),
    "connection_failure":         ("connection refused", "connection failure"),
    "zero_connections_state":     ("zero connections", "no connections"),
    "complete_service_failure":   ("service down", "service failure"),
    "memory_leak_gc_pressure":    ("memory leak", "gc pressure", "gc pause"),
    "response_latency_spike":     ("latency spike", "p99", "high latency"),
    "timeout_cascade":            ("timeout cascad", "retry storm"),
    "circuit_breaker_open":       ("circuit breaker", "breaker open"),
}


def _observed_effects(scenario: dict[str, Any]) -> dict[str, list[str]]:
    """Mine (service -> [effect_strings]) from public log haystack.

    Reads only ``scenario["logs"]`` — never ``scenario["chain"]``. Returns
    a dict keyed by service name with the canonical effect strings whose
    keywords appear in that service's logs. Order is preserved by first-
    appearance so the policy sees a stable action menu across resets.
    """
    out: dict[str, list[str]] = {}
    logs = scenario.get("logs", {}) or {}
    for service, entries in logs.items():
        seen: list[str] = []
        joined = " ".join(
            (e.get("message") or "").lower() for e in (entries or [])
        )
        for effect, keywords in _EFFECT_KEYWORDS.items():
            if any(kw in joined for kw in keywords) and effect not in seen:
                seen.append(effect)
        if seen:
            out[service] = seen[:3]  # cap per service so menu stays bounded
    return out
F. Generate chain-bearing SUBMIT candidates.
Modify _candidate_actions. Currently
lines 120-125 emit one submit per cause
candidate with final_chain=[]. After computing effects_per_service, also
emit chain-bearing variants. To keep the menu bounded (≤ ~150 actions for
500-episode REINFORCE to converge), generate top-K chain shapes rather
than the full Cartesian:

effects_per_service = _observed_effects(scenario)

# Build up to ``CHAIN_K`` distinct chain shapes per scenario by walking the
# service graph from the failing leaf upward. Order: target service first,
# then its dependents (reading scenario["service_graph"]). This matches how
# the seed_generator constructs ground-truth chains (target -> upstreams).
CHAIN_K = 4
chain_candidates: list[list[dict[str, str]]] = []
graph = scenario.get("service_graph", {}) or {}
# Heuristic ordering: services with effects in their logs, sorted by
# error_rate_during_incident if available (failing service first).
svc_meta = {s["name"]: s for s in (scenario.get("services") or [])}
ordered_svcs = sorted(
    effects_per_service.keys(),
    key=lambda s: -float(svc_meta.get(s, {}).get("error_rate_during_incident", 0.0)),
)
for svc in ordered_svcs[:CHAIN_K]:
    effects = effects_per_service.get(svc, [])
    if not effects:
        continue
    # One chain per (lead-effect) — keep each chain short (2-3 nodes) to
    # match the typical ground-truth chain length the seed_generator uses.
    chain = [{"service": svc, "effect": effects[0]}]
    for other in ordered_svcs:
        if other == svc:
            continue
        oe = effects_per_service.get(other, [])
        if oe:
            chain.append({"service": other, "effect": oe[0]})
        if len(chain) >= 3:
            break
    chain_candidates.append(chain)

# Emit chain-bearing SUBMIT actions paired with each cause candidate.
# This grows the menu by len(cause_candidates) * len(chain_candidates),
# typically ~6-12 * 4 = 24-48 extra actions — well within the 50-200 budget
# the REINFORCE convergence comment at line 543 calls out.
for cand in cause_candidates:
    for chain in chain_candidates:
        actions.append({
            "action_type": ActionType.SUBMIT.value,
            "final_cause": cand,
            "final_chain": chain,
        })
